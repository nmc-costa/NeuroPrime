def plot_between_subjects(group_l=['CG', 'EG'], sheet_l=['alpha'], 
                          feat_l=['ratio_power_threshold'], 
                          protocol='eo', select_subjects=False, exclude_subjects=True,
                          save=False, plot_stats=True, stats_test='ttest_pair', 
                          scale=10, stats_loc=0.3, group_sufix='allsubjects'):
    #group to plot
    group_l=group_l # ["CG", "EG"], ["CG"], ["EG"]
    #protocol to plot
    protocol=protocol
    #sheets to plot
    sheet_l=sheet_l #['alpha']-one band or ['theta','alpha','SMR','beta']-multiple band
    sheet_df_d={} #sheet df dict
    for sheet in sheet_l:
        sheet_df_d[sheet] = pd.read_excel(database_path, sheet_name=sheet)
        #select subjects - using 'eeg_alpha_groups'
        sheet_df=sheet_df_d[sheet]
        select_subjects=select_subjects
        group_sufix=group_sufix#default
        if select_subjects and not group_sufix=='allsubjects':
            group_sufix=group_sufix
            group=protocol+'_'+group_sufix #group of subjects
            subjects_l=eeg_alpha_groups['subject'].loc[(eeg_alpha_groups[group]==1)].tolist() #select subjects to list
            sheet_df=sheet_df.loc[sheet_df['subject'].isin(subjects_l)] #get alpha features dataframe of subjects
        exclude_subjects=exclude_subjects
        if exclude_subjects:
            group=protocol+'_EX_SUB' #group of subjects
            subjects_l=eeg_alpha_groups['subject'].loc[(eeg_alpha_groups[group]==1)].tolist() #select subjects to list
            sheet_df=sheet_df.loc[sheet_df['subject'].isin(subjects_l) != True]    #print([alpha_df['subject'].isin(subjects_l) != True])
        sheet_df_d[sheet]=sheet_df #substitute sheet_df
    #feature to plot
    feat_l=feat_l #'ratio_power_threshold', 'tBAT', 'tonic_increase', 'tonic_increase_BAT'
    #tasks to plot- [("x axis name","tasks to use")]
    task_l=[("Block_in",['1']), ("Block_out",['2'])]

    #SAVE - file_sufix to add to filename and plot
    file_sufix=protocol+'_'+feat_l[0]+'_'+group_sufix
    save=save

    #PLOT STATs
    plot_stats=plot_stats
    task_ref_stats=task_l[0][0]#reference task to make stats
    #1) one-sample paired ttest: 1_measurement_var=tBAT, 1_nominal_var=subjects, 2_nominal_var=task_out-task_ref ; hipotheses: ; 
    # biological question: Whether the measurment var has gone up or down between task_out and task_ref.
    #null hypothesis: mean difference between paired observations is zero.
    #2) two-sample ttest (independent or dependent samples): 1_measurement_var=tBAT, 1_nominal_var=Task=['Task_out' vs 'Task_ref'];
    #assumption: normally distibuted; null hypothesis:  means of the measurement var are equal for the two categories.
    #http://www.biostathandbook.com/pairedttest.html
    stats_test=stats_test#"ttest_pair"; 'ttest_ind' , 'ttest_rel'
    scale=scale
    stats_loc=stats_loc#localization of stats
    e2_f.plot_group(datadir, save=save,group_l=group_l,task_l=task_l, sheet_l=sheet_l, sheet_df_d=sheet_df_d, feat_l=feat_l, file_sufix=file_sufix, uplims=True, lolims=True, plot_stats=plot_stats, task_ref_stats=task_ref_stats,stats_test=stats_test,stats_loc=stats_loc, scale=scale)
